imputed_data <- data 
for (i in 1:nrow(imputed_data)) {
    if (is.na(imputed_data$steps[i])) {
        imputed_data$steps[i] <- mean_steps[which(imputed_data$interval[i] == mean_steps$interval), ]$avg_steps
    }
